To register this database as a datasource in your application, insert the following XML snippets into your application configuration files.

WEB-INF/stax-web.xml (or META-INF/stax-application.xml)

<resource name="jdbc/liftinaction" auth="Container" type="javax.sql.DataSource">
 <param name="username" value="manning" />
 <param name="password" value="perrett" />
 <param name="url" value="jdbc:stax://liftinaction" />
</resource>
		
WEB-INF/web.xml

<resource-ref>
 <res-ref-name>jdbc/liftinaction</res-ref-name>
 <res-type>javax.sql.DataSource</res-type>
 <res-auth>Container</res-auth>
</resource-ref>
